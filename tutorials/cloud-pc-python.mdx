## Overview

A Cloud PC is an E2B-backed virtual desktop that the Skygen agent controls remotely. In this tutorial, you will:

1. Authenticate with the Skygen API
2. Start a Cloud PC session
3. Send an agent task that performs GUI automation on the Cloud PC
4. Stream the agent's progress via Socket.IO
5. Retrieve screenshots captured during the task

Total time: approximately 30 minutes.

---

## Step 1: Authenticate

```python
import httpx

BASE_URL = "https://api.skygen.ai"

# Option A: Use your existing JWT token
TOKEN = "your-skygen-jwt-token"

# Option B: Log in to obtain a token
resp = httpx.post(
    f"{BASE_URL}/auth/login",
    json={"email": "you@example.com", "password": "your-password"},
)
TOKEN = resp.json()["access_token"]

headers = {
    "Authorization": f"Bearer {TOKEN}",
    "Content-Type": "application/json",
}

# Verify auth
me = httpx.get(f"{BASE_URL}/auth/me", headers=headers).json()
print(f"Authenticated as: {me['email']}")
```

## Step 3: Create a chat session

All agent tasks run inside a chat session. Create one:

```python
resp = httpx.post(
    f"{BASE_URL}/chat/sessions",
    headers=headers,
    json={"title": "Cloud PC Automation Task"},
)
session = resp.json()
SESSION_ID = session["id"]
print(f"Session created: {SESSION_ID}")
```

## Step 4: Set up real-time progress monitoring

Before sending the task, connect to Socket.IO to stream progress:

```python
import asyncio
import socketio

sio = socketio.AsyncClient()
progress_events = []

@sio.on("chat_message", namespace="/user")
async def on_chat_message(data):
    meta = data.get("meta_data", {})
    msg_type = meta.get("type")
    if msg_type == "ui_meta_update":
        ui_meta = meta.get("ui_meta", {})
        action = ui_meta.get("action_type", "")
        title = ui_meta.get("title", "")
        print(f"  [{action}] {title}")
        progress_events.append(data)
    elif data.get("role") == "assistant":
        content = data.get("content", "")
        if content:
            print(f"\nAgent: {content[:200]}")

@sio.on("session_updated", namespace="/user")
async def on_session_updated(data):
    waiting = data.get("meta_data", {}).get("waiting_for_input", False)
    if waiting:
        print("Agent is waiting for user input (ask_human)")

async def connect_socket():
    await sio.connect(
        BASE_URL,
        auth={"token": TOKEN},
        namespaces=["/user"],
        transports=["websocket"],
    )
    print("Socket.IO connected")

asyncio.run(connect_socket())
```

## Step 5: Send the automation task

Send a task that instructs the agent to use the Cloud PC:

```python
TASK_PROMPT = """
Use the Cloud PC to complete the following task:

1. Open a web browser (Firefox or Chromium)
2. Navigate to https://news.ycombinator.com
3. Take a screenshot of the front page
4. Extract the titles of the top 5 stories
5. Open a text editor (gedit or mousepad)
6. Type the 5 story titles as a numbered list
7. Take a final screenshot showing the text editor

Report back with the 5 story titles.
"""

resp = httpx.post(
    f"{BASE_URL}/chat/sessions/{SESSION_ID}/messages",
    headers=headers,
    json={
        "content": TASK_PROMPT,
        "mode": "cua",  # Computer Use Agent mode
    },
)
print(f"Task sent: {resp.status_code}")
print(f"Watch live at: https://app.skygen.ai/chat/{SESSION_ID}")
```

## Step 6: Wait for task completion

Poll the session status until the agent finishes:

```python
import time

def wait_for_completion(session_id: str, timeout: int = 300) -> dict:
    """Poll for task completion. Returns the final session state."""
    start = time.time()
    while time.time() - start < timeout:
        resp = httpx.get(
            f"{BASE_URL}/chat/sessions/{session_id}",
            headers=headers,
        )
        session = resp.json()
        meta = session.get("meta_data", {})
        
        status = meta.get("task_status")
        waiting = meta.get("waiting_for_input", False)
        
        print(f"Status: {status}, waiting_for_input: {waiting}")
        
        if status in ("completed", "failed", "cancelled"):
            return session
        if waiting:
            print("Agent needs input — check the web UI")
            return session
        
        time.sleep(5)
    
    raise TimeoutError(f"Task did not complete within {timeout}s")

final_session = wait_for_completion(SESSION_ID)
print(f"Task finished with status: {final_session['meta_data'].get('task_status')}")
```

## Step 7: Retrieve screenshots

After the task completes, fetch all screenshots captured by the agent:

```python
resp = httpx.get(
    f"{BASE_URL}/screenshots/agent/session/{SESSION_ID}",
    headers=headers,
)
screenshots = resp.json()
print(f"Captured {screenshots['count']} screenshots:")

for shot in screenshots["screenshots"]:
    print(f"  Step {shot['step_number']}: {shot['url']} ({shot['size']} bytes)")
```

### Download a screenshot

```python
import os

os.makedirs("screenshots", exist_ok=True)

for shot in screenshots["screenshots"]:
    img_resp = httpx.get(
        f"{BASE_URL}{shot['url']}",
        headers=headers,
    )
    filename = f"screenshots/step_{shot['step_number']:04d}.png"
    with open(filename, "wb") as f:
        f.write(img_resp.content)
    print(f"Downloaded: {filename}")
```

## Complete example

Here is the complete script combining all steps:

```python
import asyncio
import httpx
import time
import os

BASE_URL = "https://api.skygen.ai"
TOKEN = "your-skygen-jwt-token"
headers = {"Authorization": f"Bearer {TOKEN}", "Content-Type": "application/json"}

def create_session() -> str:
    resp = httpx.post(f"{BASE_URL}/chat/sessions", headers=headers,
                      json={"title": "HN Scraper"})
    return resp.json()["id"]

def send_task(session_id: str) -> None:
    httpx.post(f"{BASE_URL}/chat/sessions/{session_id}/messages",
               headers=headers,
               json={"content": "Open Firefox, go to news.ycombinator.com, "
                                 "screenshot it, list the top 5 titles.",
                     "mode": "cua"})

def wait(session_id: str, timeout: int = 300) -> None:
    start = time.time()
    while time.time() - start < timeout:
        s = httpx.get(f"{BASE_URL}/chat/sessions/{session_id}",
                      headers=headers).json()
        status = s.get("meta_data", {}).get("task_status")
        if status in ("completed", "failed"):
            print(f"Done: {status}")
            return
        time.sleep(5)

def get_screenshots(session_id: str) -> None:
    data = httpx.get(f"{BASE_URL}/screenshots/agent/session/{session_id}",
                     headers=headers).json()
    os.makedirs("output", exist_ok=True)
    for s in data["screenshots"]:
        img = httpx.get(f"{BASE_URL}{s['url']}", headers=headers)
        fn = f"output/step_{s['step_number']:04d}.png"
        with open(fn, "wb") as f:
            f.write(img.content)
        print(f"Saved: {fn}")

if __name__ == "__main__":
    sid = create_session()
    print(f"Session: {sid}")
    send_task(sid)
    wait(sid)
    get_screenshots(sid)
    print("Done!")
```

## Troubleshooting

**Agent asks to start Cloud PC**  
Your Cloud PC is not running. Click "Start Cloud PC" in the ask_human prompt or provision one via `POST /cloud-pc/start`. The agent will resume automatically when the PC is ready.

**Task times out**  
Increase the `timeout` parameter. Complex GUI tasks on a fresh Cloud PC (browser startup, page load) can take 2–3 minutes. The default 300-second timeout is usually sufficient.

**Screenshots are empty**  
The agent may not have reached the screenshot step. Check the action history via `GET /action-history?session_id={id}` to see where it stopped.

**Permission denied on proxy URL**  
Ensure you include the `Authorization` header when downloading screenshots from `/screenshots/proxy/`. The proxy enforces ownership.
